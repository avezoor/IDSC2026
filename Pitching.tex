\begin{frame}[t]{The Healthcare Problem}
\begin{columns}[T,onlytextwidth]
\column{0.58\textwidth}
\justifying
\begin{block}{Why this matters}
\begin{itemize}
  \item Brugada syndrome can lead to sudden cardiac death, but its ECG pattern can be subtle.
  \item In screening, a false negative is more dangerous than an extra false alarm.
  \item Many AI studies chase accuracy, even when the real clinical question is: \textbf{who are we still missing?}
\end{itemize}
\end{block}

\begin{alertblock}{Our pitch}
Use ordinary 12-lead ECGs to build a clinically honest screening aid that helps specialists miss fewer dangerous cases.
\end{alertblock}

\begin{frame}[t]{Our Mathematical and Computational Solution}
\small
\begin{columns}[T,onlytextwidth]
\column{0.55\textwidth}
\begin{block}{Pipeline}
\begin{enumerate}
  \item Start from raw WFDB 12-lead ECGs.
  \item Build median-beat and engineered-feature representations.
  \item Train 11 complementary models on the same patient-level split.
  \item Tune thresholds on validation data, then test once.
\end{enumerate}
\end{block}

\begin{exampleblock}{Clinical design choice}
The primary benchmark is \textbf{ECG-only}. Metadata stays in a separate ablation to avoid shortcut learning.
\end{exampleblock}

\column{0.42\textwidth}
\begin{block}{Safety-aware ranking}
\[
\begin{aligned}
\text{Composite} ={}& 0.40P + 0.35S + 0.15C \\
&+ 0.05E + 0.05T
\end{aligned}
\]
\begin{itemize}
  \item $P$: performance
  \item $S$: safety
  \item $C$: consistency
  \item $E$: efficiency
  \item $T$: stability evidence
\end{itemize}
\end{block}

\begin{alertblock}{Key idea}
We do not let raw leaderboard score hide a poor false-negative profile.
\end{alertblock}
\end{columns}
\end{frame}

% Suggested speaking time: 1:25--1:50
\begin{frame}[t]{Why 11 Models Matter for Healthcare Hope}
\scriptsize
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Feature-based models}
\begin{itemize}
  \item \textbf{XGBoost}: strong nonlinear baseline on engineered ECG biomarkers.
  \item \textbf{Random Forest}: robust ensemble for noisy and variable patient signals.
  \item \textbf{SVM-RBF}: captures subtle curved decision boundaries in borderline ECGs.
  \item \textbf{AdaBoost}: focuses learning on the hardest-to-classify cases.
\end{itemize}
\end{block}

\begin{exampleblock}{End-to-end deep models}
\begin{itemize}
  \item \textbf{1D CNN Median Beat}: learns local waveform cues directly from ECG morphology.
  \item \textbf{ResNet-1D Median Beat}: goes deeper without unstable training, improving safety-oriented detection.
  \item \textbf{CNN + BiGRU Median Beat}: combines local patterns with sequential beat context.
  \item \textbf{Transformer Encoder Median Beat}: models wider lead-to-lead and time interactions.
\end{itemize}
\end{exampleblock}

\column{0.48\textwidth}
\begin{alertblock}{Transfer and advanced models}
\begin{itemize}
  \item \textbf{Transfer Learning (In-domain DAE to Classifier)}: reuses ECG structure learned before classification, helping when labeled data is limited.
  \item \textbf{VICReg (Self-supervised)}: learns useful signal representations without needing labels on every step.
  \item \textbf{Echo State Network (ESN)}: provides a lightweight dynamic sequence baseline for efficient screening.
\end{itemize}
\end{alertblock}

\begin{beamercolorbox}[wd=\linewidth,rounded=true,sep=0.8em]{block body}
\textbf{Why this brings hope:} we did not trust one lucky algorithm. We tested 11 mathematical views of the same ECG so clinicians get a more credible, safety-aware, and practical screening aid.
\end{beamercolorbox}
\end{columns}
\end{frame}

\begin{frame}[t]{What the Benchmark Found}

  \item PR-AUC: 0.694, F1: 0.696
  \item Sensitivity: 0.571, Specificity: 0.983
  \item False Negatives: 6
  \item Selected by the balanced multi-aspect ranking

\textbf{ResNet-1D Median Beat}
\begin{itemize}

  \item False Negatives: 5
  \item Better when missing a Brugada case is the main concern
\end{itemize}
